#include <LiquidCrystal.h>

const int numRows = 4;
const int numCols = 20;

LiquidCrystal lcd(12, 11, 5, 4, 3, 2);

int currentOption = 0;

String options[] = {
  "Play Game",
  "Instructions",
  "High Scores",
  "Settings"
};

void setup() {
  lcd.begin(numCols, numRows);
  lcd.clear();
}

void loop() {
  // Display the menu
  lcd.setCursor(0, 0);
  lcd.print("--- Main Menu ---");
  for (int i = 0; i < numRows - 1; i++) {
    lcd.setCursor(0, i + 1);
    if (i == currentOption) {
      lcd.print("> ");
    } else {
      lcd.print("  ");
    }
    lcd.print(options[i]);
  }

  // Handle user input
  if (digitalRead(LEFT_BUTTON) == LOW) {
    // Move to previous option
    currentOption--;
    if (currentOption < 0) {
      currentOption = numRows - 2;
    }
  }
  if (digitalRead(RIGHT_BUTTON) == LOW) {
    // Move to next option
    currentOption++;
    if (currentOption >= numRows - 1) {
      currentOption = 0;
    }
  }
  if (digitalRead(SELECT_BUTTON) == LOW) {
    // Select current option
    switch (currentOption) {
      case 0:
        // Play game
        break;
      case 1:
        // Show instructions
        break;
      case 2:
        // Show high scores
        break;
      case 3:
        // Show settings
        break;
    }
  }
}
